SHOW databases;

CREATE DATABASE projectdb;

-- ユーザ情報
CREATE TABLE IF NOT EXISTS USER_DATA (
  -- ユーザID
  SEQ_USER_ID BIGINT AUTO_INCREMENT NOT NULL PRIMARY KEY COMMENT 'ユーザID',
  -- パスワード
  PASSWORD VARCHAR(64) NOT NULL COMMENT 'パスワード',
  -- 登録日時
  REG_DATE DATETIME NOT NULL COMMENT '登録日時',
  -- 更新日時
  UPDATE_DATE DATETIME NOT NULL COMMENT '更新日時'
);

-- 日別ユーザ勤怠情報
CREATE TABLE IF NOT EXISTS DAILY_USER_WORK_DATA (
  -- 日別ユーザ勤怠情報ID
  DAILY_USER_WORK_DATA_ID BIGINT AUTO_INCREMENT NOT NULL PRIMARY KEY COMMENT '日別ユーザ勤怠情報ID',
  -- ユーザID
  SEQ_USER_ID BIGINT NOT NULL COMMENT 'ユーザID',
  -- 企業コード
  COMPANY_CODE VARCHAR(16) NOT NULL COMMENT '企業コード',
  -- 勤怠情報登録日時
  WORK_DATA_REG_DATE DATETIME NOT NULL COMMENT '勤怠情報登録日時',
  -- 始業時刻
  WORK_START_TIME VARCHAR(4) NOT NULL COMMENT '始業時刻',
  -- 終業時刻
  WORK_END_DATE VARCHAR(4) NOT NULL COMMENT '終業時刻',
  -- 実労働時間
  ACTUAL_WORK_TIME VARCHAR(4) NOT NULL COMMENT '実労働時間',
  -- 登録日時
  REG_DATE DATETIME NOT NULL COMMENT '登録日時',
  -- 更新日時
  UPDATE_DATE DATETIME NOT NULL COMMENT '更新日時'
);
